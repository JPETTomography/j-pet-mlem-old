      --s-z             TOF sigma along z axis (float [=0.015])

  -r, --radius          inner detector ring radius (float [=auto])

      --radius3          ... 3rd ring (float [=auto])
      --radius4          ... 4th ring (float [=auto])
      --rotation        ring rotation (0-1) (float [=auto])
      --rotation2        ... 2nd ring (float [=auto])
      --rotation3        ... 3rd ring (float [=auto])
      --rotation4        ... 4th ring (float [=auto])
  -d, --n-detectors     number of detectors in ring (int [=0])
      --n-detectors2     ... 2nd ring (int [=0])
      --n-detectors3     ... 3rd ring (int [=0])
      --n-detectors4     ... 4th ring (int [=0])
      --fov-radius      field of view radius (float [=auto])
  -e, --n-emissions     number of emissions (int [=0])
  -p, --s-pixel         pixel size (float [=auto])

      --scale           Scale phantom with given constant (float [=1])

  -o, --output          output lor hits for supplied phantom (file [=])
      --detected        collects detected emissions
  -v, --verbose         prints the iterations information on std::out
  -S, --seed            random number generator seed (unsigned int [=0])
  -T, --n-threads       number of OpenMP threads (int [=0])
  -?, --help            print this message

  --w-detector, --d-detector, --n-detectors
